int * binary_addition_gen(int *c1, int *c2){
static int bnr[32];
int carry=0,i;

for(i=0;i<=31;i++){
if(carry==0){
  if(*(c1+i)==0&&*(c2+i)==0){
   bnr[i]=0;
   carry=0;
  }
  if((*(c1+i)==1&&*(c2+i)==0)||(*(c1+i)==0&&*(c2+i)==1)){
   bnr[i]=1;
   carry=0;
  }
  if(*(c1+i)==1&&*(c2+i)==1){
   bnr[i]=0;
   carry=1;
  }
}
else if(carry==1){
  if(*(c1+i)==0&&*(c2+i)==0){
   bnr[i]=1;
   carry=0;
  }
  if((*(c1+i)==1&&*(c2+i)==0)||(*(c1+i)==0&&*(c2+i)==1)){
   bnr[i]=0;
   carry=1;
  }
  if(*(c1+i)==1&&*(c2+i)==1){
   bnr[i]=1;
   carry=1;
  }
}
}
return bnr;
}
